Variable as argument

cat myfile.json \
    | jq --arg foo bar '.[] | select(.name == $foo)'

Multiline query

echo $releases \
    | jq '.[] | select(.name == "v1.16.3").assets[]
              | select(.name == "kops-darwin-amd64").id'

Select where name contains

aws route53 list-resource-record-sets --hosted-zone-id ZIP7RXUYMBTT9 > record_sets
cat record_sets | jq '.ResourceRecordSets[]
    | select(.Name | contains("tcdaily-uswest1"))
    | .Name'

aws --region us-east-1 lambda list-functions \
    | jq '.Functions[] | select(.FunctionName | startswith("streaming-ingest"))'

Use aws –filter

aws ec2 describe-addresses --filters "Name=tag:Name,Values=K8S_EIP_POOL"

Key contains period

kubectl get secret default-token-mxxl9 -o json \
    | jq -r '.data["cloudconnect.crt"]' \
    | base64 --decode

kubectl get secret -o json \
    | jq '.items[]
          | select(.metadata.annotations["kubernetes.io/service-account.name"] == "default")'
